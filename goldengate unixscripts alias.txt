
--monitor the amount of memory used by the Oracle GoldenGate processes.

sh ps -C replicat -O rss | awk '{print $2/1024, "MB", $12}' | sort -k 2

--monitor the amount of virtual memory used by the Oracle GoldenGate processes.

sh  ps -C replicat -O vsz | awk '{print $2/1024, "MB", $12}' | sort -k 2
